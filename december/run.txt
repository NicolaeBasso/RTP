# Start the Message Broker
{:ok, broker} = Broker.start_link
{:ok, broker} = Message.Broker.start_link

# Start the Subscriber
{:ok, subscriber1} = Subscriber.start_link
{:ok, subscriber2} = Subscriber.start_link
{:ok, subscriber3} = Subscriber.start_link

# Subscribe the Subscribers to the Message Broker
send(broker, {:subscribe, subscriber1})
send(broker, {:subscribe, subscriber2})
send(broker, {:subscribe, subscriber3})

# Start the Publisher
publisher = Publisher

# Publish some messages
publisher.publish(broker, "Hello World!")
publisher.publish(broker, "Goodbye World!")
